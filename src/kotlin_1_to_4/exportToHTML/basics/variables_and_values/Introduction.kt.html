5     *

9     * val -> immutable 
10    * 
11    *

14   fun main(args: Array<String>) 
15   { 
16   // Immutable variable of type String that is my name! 
17       val myName = "Paul is a nice guy." 
18    
19       // val is the equivalent of a "read only" variable, it cannot be re-assigned but member variables can be changed. 
20       // consider. i.e. we can run methods and other things on it, but what we can't do is re-assign it! 
21    
22       println(myName.split(" ")) 
23    
24       // But this throws an error 
25       // myName = "Sammy" 
26    
27   // Mutable variable of type Int that is my age ... sadly. *sniff sniff* 
28       var age = 27

30
