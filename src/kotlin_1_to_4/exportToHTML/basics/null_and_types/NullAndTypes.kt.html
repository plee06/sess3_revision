13   // Type checking and casting in Kotlin is done with the *is operator 
14    
15   fun isString(str: Any): Boolean 
16   { 
17       return if (str is String) true else false 
18   } 
19    
20   // ClassCastException thrown if not 
21    
22   /* 
23   * Smart Casts in Kotlin. 
24   * 
25   * Unlike Java Kotlin is a bit smarter in "Smart casting" 
26   * 
27   * In Java we always have to explicitly cast something after checking if it instanceof for some element. 
28   * 
29   * i.e. if (str instanceof String) { String str = (String) str ... } 
30   * In Kotlin, the explicit cast is done implicitly. 
31   * 
32   * 
33   * */ 
34    
35   // Consider: 
36    
37   // It can be of type Any, but Kotlin will "smart cast" and instead of throwing an error, it says "okay" 
38   // if what you pass in is a String I'll print out the length, otherwise I won't 
39    
40   // Important to note that Smart casts will only work with variables that will not change between 
41   // time it's checked and time when it is used, i.e. mutating any local var breaks this... 
42    
43    
44   fun printIfString(s: Any): Unit 
45   { 
46       if (s is String) println(s.length) 
47   } 
48    
49   // Explicit Casting is also available in Kotlin, used with the *as operator 
50    
51   // Important to note that if a possible null is to be returned, use safe operator otherwise, ClassCastException is 
52   // thrown. 
53    
54   fun tryCast(t: Any): String? 
55   { 
56       return if (!(t is String)) t as String else null 
57   }
